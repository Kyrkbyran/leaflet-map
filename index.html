<!DOCTYPE html>

<html lang="sv">

<head>

  <meta charset="utf-8">

  <title>Min karta</title>

  <meta name="viewport" content="width=device-width, initial-scale=1.0">


  <!-- Leaflet CSS -->

  <link rel="stylesheet" href="https://unpkg.com/leaflet/dist/leaflet.css"/>


  <style>

    /* Gör så kartan fyller hela sidan */

    html, body, #map {

      height: 100%;

      margin: 0;

      padding: 0;

    }

  </style>

</head>

<body>

  <div id="map"></div>


  <!-- Leaflet JS -->

  <script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>

  <script>

    // Skapa kartan och centrera på Stockholm

    var map = L.map('map').setView([59.3293, 18.0686], 13);


    // Lägg till OpenStreetMap tiles

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

      attribution: '&copy; OpenStreetMap contributors'

    }).addTo(map);


    // Försök hitta användarens plats

    map.locate({setView: true, maxZoom: 16});


    // Visa markör på användarens plats

    map.on('locationfound', function(e) {

      L.marker(e.latlng).addTo(map)

        .bindPopup("Du är här!").openPopup();

    });


    // Hantera fel

    map.on('locationerror', function(e) {

      alert("Kunde inte hitta din plats: " + e.message);

    });

  </script>

</body>

</html>
